110void aux_undo(JOGO *game){
111 // Acede à última jogada
112 (game->jogadas_historial)--;
113 struct JOGADA ultimo_mov = game->historial[game->jogadas_historial];
114
115 int origem = ultimo_mov.pilha;
116 int chegada = ultimo_mov.chegada;
117 int n = ultimo_mov.n;
118 int pos_original = ultimo_mov.coluna; // Onde elas estavam na origem
119
120 // Localiza as cartas na chegada
121 int pos_atual_chegada = game->tamanho_pilha[chegada] - n;
122
123 //O Loop de Devolução
124 for (int i = 0; i < n; i++) {
125 game->matriz[origem][pos_original + i] = game->matriz[chegada][pos_atual_chegada + i];
126 }
127
128 // Ajustar os contadores de tamanho
129 game->tamanho_pilha[origem] += n;
130 game->tamanho_pilha[chegada] -= n;
131}
