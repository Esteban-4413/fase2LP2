308/*
309 Função Auxiliar da define_jogAtual, responsável por chamar a função que calcula o tamanho da sequência (tamanho_sequencia)
310 e com isso, já verifica se existem cartas consecutivas do mesmo naipe empilhadas e quantas são, e se a carta clicada é inválida
311 i.e. é uma carta que não está no topo, logo não pode ser movida.

313 Bool - True(1) se a jogada é valida ;

317int evalida_tamanhoSeq(JOGO *game){

342int verifica_foudations(JOGO *game){

344 if (pilha != -1){

346 else if (game->matriz[pilha][game->tamanho_pilha[pilha]-1].naipe == 'E') game->foundations[1] = 1;
347 else if (game->matriz[pilha][game->tamanho_pilha[pilha]-1].naipe == 'O') game->foundations[2] = 1;
348 else if (game->matriz[pilha][game->tamanho_pilha[pilha]-1].naipe == 'P') game->foundations[3] = 1;
349 game->tamanho_pilha[pilha] -= 13; // Tira a sequência de 13 cartas da pilha;
350 return pilha;
351 }
352 return -1;
