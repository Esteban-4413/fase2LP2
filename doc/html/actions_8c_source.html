48int valida_jogada_destino(JOGO *game){

68int tamanho_sequencia(int x, int y, JOGO *game){
69 // Condição de paragem: Se chegou à última carta da pilha, o tamanho é 1.
70 if (x == game->tamanho_pilha[y] - 1){
71 return 1;
72 }
73 // Verifica se a próxima carta é sequencial (valor - 1) e tem o mesmo naipe.
74 if ((game->matriz[y][x].valor == game->matriz[y][x+1].valor + 1)
75 && game->matriz[y][x].naipe == game->matriz[y][x+1].naipe){
76 return 1 + tamanho_sequencia(x+1, y, game);
77 }
78 return 1;
79}
